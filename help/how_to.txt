[BUILD DOCKER]
EC2-Setup
    ssh -i "your-key.pem" ubuntu@<EC2_PUBLIC_IP>
    sudo apt update
    sudo apt install -y docker.io
    sudo systemctl enable --now docker
    mkdir -p ~/todolist_data
    cd ~/todolist
    sudo docker build -t fastapi-todo .
    sudo docker ps -a
    sudo docker stop fastapi-todo
    sudo docker rm fastapi-todo
    sudo docker run -d -p 80:80 --name fastapi-todo -v ~/todolist_data:/app --restart unless-stopped fastapi-todo
